// 在函数外部添加防抖定时器变量（放在 saveAllDataToLocalStorage 函数之前） let syncTimer = null; function saveAllDataToLocalStorage() { try { const dataToSave = { chineseErrors: chineseErrors, mathErrors: mathErrors, version: '1.0', lastSaved: new Date().toISOString() }; localStorage.setItem('cuoti_data', JSON.stringify(dataToSave)); console.log('✅ 数据已保存到本地'); // ========== 防抖静默同步（5秒）========== if (GITHUB_CONFIG.token && GIST_CONFIG.id) { // 清除之前的定时器 if (syncTimer) clearTimeout(syncTimer); // 开始旋转动画（表示即将同步） startUploadSpin(); // 5秒后执行同步 syncTimer = setTimeout(() => { console.log('🔄 防抖触发，开始静默同步...'); saveToCloud().catch(e => console.log('云端同步失败:', e)).finally(() => { // 同步完成后停止旋转 setTimeout(() => stopUploadSpin(), 500); }); syncTimer = null; }, 5000); // 5秒防抖 } } catch (e) { if (e.name === 'QuotaExceededError') { alert('存储空间不足！图片太多，请删除一些旧错题。'); } else { console.error('保存失败:', e); } } }
